const checkAuth = (req, res, next) => {
const token = req.headers.authorization?.split(' ')[1];
if (!token) {
    return res.status(httpStatusCodes.unauthorized).json(
        getErrorResponseObject(null, 'No se proporcionó token')
    );
}
next();
};
